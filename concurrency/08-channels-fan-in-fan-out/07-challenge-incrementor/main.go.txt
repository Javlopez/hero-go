package main

import (
	"fmt"
	"sync"
	"sync/atomic"
)

var (
	counter int64
	wg      sync.WaitGroup
)

func main() {

	wg.Add(2)
	go Incrementor("1")
	go Incrementor("2")
	wg.Wait()
	fmt.Println("Final Counter:", counter)
}

func Incrementor(label string) {

	for i := 0; i < 20; i++ {
		atomic.AddInt64(&counter, 1)
		fmt.Printf("Process %s printing %d\n", label, i)
	}
	wg.Done()
}
